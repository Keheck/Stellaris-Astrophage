namespace = astrophage

# General chain of events:
# 1. Stars are dimming galaxy wide
# 2. Contunued dimming reduces energy and food output from planets
# 2.1 Starts a situation, with progressing stages exacerbating the penalty
# 3. Empires look for ways to create food and energy from other sources than solar
# 3.1 Choices include falling back to nuclear fission at the cost of minerals, or rationing food at the cost of job efficiency and happiness
# 4. Certain stars exhibit immunity to the Astrophage (bias towards certain system types?)
# 5. A way to "vaccinate" stars against the Astrophage is developed
# 6. Empires can choose to use or erradicate the Astrophage

# Schedule crisis to happen
event = {
	id = astrophage.1
	is_triggered_only = yes
	fire_only_once = yes
	hide_window = yes

	trigger = {
		OR = {
			mid_game_years_passed >= 0
			has_global_flag = force_astrophage_crisis
		}
	}

	immediate = {
		set_global_flag = astrophage_crisis_scheduled
		log = "Astrophage fired..."

		every_country = {
			limit = {
				is_country_type = default
			}

			if = {
				limit = { has_global_flag = force_astrophage_crisis }
				log = "...by force"
				country_event = {
					id = astrophage.2
				}
			}
			else = {
				log = "...normally"
				country_event = {
					id = astrophage.2
					years = 50
					random = 25
				}
			}
		}
	}
}

# Notify countries of the dimming suns
country_event = {
	id = astrophage.2
	is_triggered_only = yes	
	title = astrophage.2.title
	desc = astrophage.2.desc
	
	option = {
		name = astrophage.2.A
		add_modifier = {
			modifier = dimming_stars_weak
		}
	}
}
